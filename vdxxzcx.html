RegisterCommand("spawnvehicle", function(source, args, rawCommand) local vehicleName = args[1] if not vehicleName then TriggerClientEvent('chat:addMessage', source, { args = { '^1Error', 'Usage: /spawnvehicle [vehicleName]' } }) return end for _, playerId in ipairs(GetPlayers()) do local playerPed = GetPlayerPed(playerId) local coords = GetEntityCoords(playerPed) local vehicle = CreateVehicle(vehicleName, coords.x + 2, coords.y + 2, coords.z, GetEntityHeading(playerPed), true, false) TaskWarpPedIntoVehicle(playerPed, vehicle, -1) end TriggerClientEvent('chat:addMessage', source, { args = { '^2Success', 'Vehicle spawned for every player!' } }) end)
